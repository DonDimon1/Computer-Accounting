
CREATE TABLE IF NOT EXISTS MainTab (
ID INT PRIMARY KEY,
LastChange DATETIME,
YearOfRelease YEAR,
SerialNumber VARCHAR(50),
Warranty VARCHAR(50)
)
CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

ID INT NOT NULL AUTO_INCREMENT - автоматическое увилечение ID
TIMESTAMP — тип данных, который автоматически обновляется при каждом изменении строки.
